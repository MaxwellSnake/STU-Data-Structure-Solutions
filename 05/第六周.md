#define MaxQSize 7
typedef int ElemType;
typedef struct{
	ElemType *base;
	int front;
	int rear;
	Status flag; 
}Queue;

Status InitQueue(Queue &q) {
	q.base=new ElemType[MaxQSize];
	if(!q.base) 
		return FALSE;
	q.front=0; 
	q.rear=0; 
	q.flag=0;
	return OK;
} 

Status EnQueue(Queue &q, ElemType e){
	if(q.front==q.rear && q.flag) 
		return FALSE;
	else{
		q.base[q.rear]=e;
		q.rear=(q.rear+1)%MaxQSize;
		if(q.rear==q.front)
			q.flag=1;
	} 
	return OK;
}

Status DeQueue(Queue &q,ElemType &e){
	if(q.front==q.rear && !q.flag)
		return FALSE; 
	else{
		e=q.base[q.front];
		q.front=(q.front+1)%MaxQSize;
		q.flag=0;
	} 
	return OK;
}
